class MyComparator implements Comparator<Track> {
	@Override
	public int compare(Track o1, Track o2) {
		if (o1.GetRadius() < o2.GetRadius()) {
			return -1;
		} else if (o1.GetRadius() > o2.GetRadius()) {
			return 1;
		} else {
			return 0;
		}
	}
